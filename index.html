#include int main() { int n; printf("Enter number of processes: "); scanf("%d", &n); int arrival[50], burst[50], completion[50], turnaround[50], waiting[50]; int processID[50]; for (int i = 0; i < n; i++) { processID[i] = i + 1; printf("Enter Arrival and Burst for P%d: ", i + 1); scanf("%d %d", &arrival[i], &burst[i]); } // Sorting by arrival time for (int i = 0; i < n - 1; i++) for (int j = 0; j < n - i - 1; j++) if (arrival[j] > arrival[j + 1]) { int t; t = arrival[j]; arrival[j] = arrival[j + 1]; arrival[j + 1] = t; t = burst[j]; burst[j] = burst[j + 1]; burst[j + 1] = t; t = processID[j]; processID[j] = processID[j + 1]; processID[j + 1] = t; } int currentTime = 0; double totalWT = 0, totalTAT = 0; for (int i = 0; i < n; i++) { if (currentTime < arrival[i]) currentTime = arrival[i]; currentTime += burst[i]; completion[i] = currentTime; turnaround[i] = completion[i] - arrival[i]; waiting[i] = turnaround[i] - burst[i]; totalWT += waiting[i]; totalTAT += turnaround[i]; } printf("\nP\tAT\tBT\tCT\tTAT\tWT\n"); for (int i = 0; i < n; i++) printf("P%d\t%d\t%d\t%d\t%d\t%d\n", processID[i], arrival[i], burst[i], completion[i], turnaround[i], waiting[i]); printf("\nAverage Waiting Time: %.2f\n", totalWT / n); printf("Average Turnaround Time: %.2f\n", totalTAT / n); return 0; }
